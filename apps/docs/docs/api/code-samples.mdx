## JavaScript/TypeScript Examples

### Basic API Client

```typescript
class AbdalkaderAPI {
  private apiKey: string;
  private baseURL: string;

  constructor(apiKey: string, baseURL = 'https://abdalkader.dev/api') {
    this.apiKey = apiKey;
    this.baseURL = baseURL;
  }

  private async request<T>(
    endpoint: string,
    options: RequestInit = {}
  ): Promise<T> {
    const url = `${this.baseURL}${endpoint}`;
    const response = await fetch(url, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        'X-API-Key': this.apiKey,
        ...options.headers,
      },
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.message || 'Request failed');
    }

    return response.json();
  }

  async getHealth() {
    return this.request('/health');
  }

  async submitPerformanceMetric(metric: {
    name: string;
    value: number;
    timestamp: number;
    url: string;
    userAgent: string;
    [key: string]: unknown;
  }) {
    return this.request('/performance', {
      method: 'POST',
      body: JSON.stringify(metric),
    });
  }

  async reportError(error: {
    message: string;
    stack?: string;
    url: string;
    userAgent: string;
    timestamp: number;
    sessionId: string;
    environment: string;
    [key: string]: unknown;
  }) {
    return this.request('/errors', {
      method: 'POST',
      body: JSON.stringify(error),
    });
  }
}

// Usage
const api = new AbdalkaderAPI(process.env.API_KEY!);
const health = await api.getHealth();
```

### React Hook

```typescript
import { useState, useEffect } from 'react';

function useAbdalkaderAPI(apiKey: string) {
  const [health, setHealth] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    async function fetchHealth() {
      try {
        const response = await fetch('https://abdalkader.dev/api/health');
        const data = await response.json();
        setHealth(data);
      } catch (err) {
        setError(err);
      } finally {
        setLoading(false);
      }
    }

    fetchHealth();
  }, []);

  const submitMetric = async (metric: {
    name: string;
    value: number;
    timestamp: number;
    url: string;
    userAgent: string;
  }) => {
    const response = await fetch('https://abdalkader.dev/api/performance', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-API-Key': apiKey,
      },
      body: JSON.stringify(metric),
    });

    return response.json();
  };

  return { health, loading, error, submitMetric };
}
```

// Largest Contentful Paint
new PerformanceObserver((list) => {

## Python Examples

### Basic API Client

```python
import requests
from typing import Dict, Any, Optional
import os

class AbdalkaderAPI:
    def __init__(self, api_key: str, base_url: str = 'https://abdalkader.dev/api'):
        self.api_key = api_key
        self.base_url = base_url
        self.headers = {
            'Content-Type': 'application/json',
            'X-API-Key': api_key
        }

    def request(
        self,
        endpoint: str,
        method: str = 'GET',
        data: Optional[Dict[str, Any]] = None
    ) -> Dict[str, Any]:
        url = f'{self.base_url}{endpoint}'
        response = requests.request(
            method=method,
            url=url,
            headers=self.headers,
            json=data
        )
        response.raise_for_status()
        return response.json()

    def get_health(self) -> Dict[str, Any]:
        return self.request('/health')

    def submit_performance_metric(self, metric: Dict[str, Any]) -> Dict[str, Any]:
        return self.request('/performance', method='POST', data=metric)

    def report_error(self, error: Dict[str, Any]) -> Dict[str, Any]:
        return self.request('/errors', method='POST', data=error)

# Usage
api = AbdalkaderAPI(os.getenv('API_KEY'))
health = api.get_health()
```

### Async API Client

```python
import aiohttp
import asyncio
from typing import Dict, Any, Optional

class AsyncAbdalkaderAPI:
    def __init__(self, api_key: str, base_url: str = 'https://abdalkader.dev/api'):
        self.api_key = api_key
        self.base_url = base_url
        self.headers = {
            'Content-Type': 'application/json',
            'X-API-Key': api_key
        }

    async def request(
        self,
        endpoint: str,
        method: str = 'GET',
        data: Optional[Dict[str, Any]] = None
    ) -> Dict[str, Any]:
        url = f'{self.base_url}{endpoint}'
        async with aiohttp.ClientSession() as session:
            async with session.request(
                method=method,
                url=url,
                headers=self.headers,
                json=data
            ) as response:
                response.raise_for_status()
                return await response.json()

    async def get_health(self) -> Dict[str, Any]:
        return await self.request('/health')

    async def submit_performance_metric(self, metric: Dict[str, Any]) -> Dict[str, Any]:
        return await self.request('/performance', method='POST', data=metric)

    async def report_error(self, error: Dict[str, Any]) -> Dict[str, Any]:
        return await self.request('/errors', method='POST', data=error)

# Usage
async def main():
    api = AsyncAbdalkaderAPI(os.getenv('API_KEY'))
    health = await api.get_health()
    print(health)

asyncio.run(main())
```

### Performance Monitoring

```python
import time
import requests
from typing import Dict, Any

class PerformanceMonitor:
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = 'https://staging.abdalkader.dev/api'
        self.headers = {
            'Content-Type': 'application/json',
            'X-API-Key': api_key
        }

    def track_api_call(
        self,
        endpoint: str,
        method: str,
        duration: float,
        url: str,
        user_agent: str
    ):
        metric = {
            'name': 'api_response_time',
            'value': int(duration * 1000),  # Convert to milliseconds
            'timestamp': int(time.time() * 1000),
            'url': url,
            'userAgent': user_agent,
            'endpoint': endpoint,
            'method': method
        }

        requests.post(
            f'{self.base_url}/performance',
            headers=self.headers,
            json=metric
        )

    def timed_request(self, endpoint: str, method: str = 'GET', **kwargs):
        start_time = time.time()
        try:
            response = requests.request(method=method, url=endpoint, **kwargs)
            duration = time.time() - start_time
            
            self.track_api_call(
                endpoint=endpoint,
                method=method,
                duration=duration,
                url=endpoint,
                user_agent='Python-Client/1.0'
            )
            
            return response
        except Exception as e:
            duration = time.time() - start_time
            self.track_api_call(
                endpoint=endpoint,
                method=method,
                duration=duration,
                url=endpoint,
                user_agent='Python-Client/1.0'
            )
            raise

# Usage
monitor = PerformanceMonitor(os.getenv('API_KEY'))
response = monitor.timed_request('https://api.example.com/data')
```

### Error Reporting

```python
import traceback
import sys
from typing import Dict, Any, Optional

class ErrorReporter:
    def __init__(self, api_key: str, session_id: str, environment: str = 'production'):
        self.api_key = api_key
        self.session_id = session_id
        self.environment = environment
        self.base_url = 'https://staging.abdalkader.dev/api'
        self.headers = {
            'Content-Type': 'application/json',
            'X-API-Key': api_key
        }

    def report_error(
        self,
        error: Exception,
        url: str,
        user_agent: str,
        additional_context: Optional[Dict[str, Any]] = None
    ):
        error_report = {
            'message': str(error),
            'stack': traceback.format_exc(),
            'url': url,
            'userAgent': user_agent,
            'timestamp': int(time.time() * 1000),
            'sessionId': self.session_id,
            'environment': self.environment,
            'additionalContext': additional_context or {}
        }

        try:
            response = requests.post(
                f'{self.base_url}/errors',
                headers=self.headers,
                json=error_report
            )
            return response.json()
        except Exception as e:
            print(f'Failed to report error: {e}')

    def setup_global_handler(self):
        def exception_handler(exc_type, exc_value, exc_traceback):
            if issubclass(exc_type, KeyboardInterrupt):
                sys.__excepthook__(exc_type, exc_value, exc_traceback)
                return

            self.report_error(
                exc_value,
                url='unknown',
                user_agent='Python-Client/1.0',
                additional_context={
                    'type': exc_type.__name__,
                    'traceback': traceback.format_exception(exc_type, exc_value, exc_traceback)
                }
            )

        sys.excepthook = exception_handler

# Usage
reporter = ErrorReporter(
    api_key=os.getenv('API_KEY'),
    session_id='sess_123',
    environment='production'
)
reporter.setup_global_handler()
```
